# Html Helper
       
    Html.ActionLink()       ->  <a></a>
    Html.TextBoxFor()       ->  <input type="textbox">
    Html.TextAreaFor()      ->  <input type="textarea">
    Html.CheckBoxFor()      ->  <input type="checkbox">
    Html.RadioButtonFor()   ->  <input type="radio">
    Html.DropDownListFor()  ->  <select><option></select>
    Html.ListBoxFor()	    ->  multi-select list box: <select></select>
    Html.HiddenFor()	    ->  <input type="hidden">
    Html.PasswordFor()	    ->  <input type="password">
    Html.DisplayFor()	    ->  HTML text: ""
    Html.LabelFor()         ->  <label></label>
    Html.EditorFor()	    ->  textbox for string property, numeric field for int, double or other numeric type.
         
    for example

        @Html.TextBoxFor(m => m.StudentName, new { @class = "form-control" })
        equals to
        <input class="form-control" id="StudentName" name="StudentName" type="text" value="" />

        @Html.CheckBoxFor(m => m.isActive)
        equals to
        <input data-val="true" id="isActive" name="isActive" type="checkbox" />

        @Html.RadioButtonFor(m => m.Gender,"Male")
        @Html.RadioButtonFor(m => m.Gender,"Female")
        equals to
        <input checked="checked" id="Gender" name="Gender" type="radio" value="Male" />
        <input id="Gender" name="Gender" type="radio" value="Female" />
