# Date

    get current date & time 
        var d = new Date();     // output -> Thu Apr 20 2023 11:50:34 GMT+0330 (Iran Standard Time)
        for convert it to miliseconds passed since 1970
            Number(d);

    declear a date
        Date("2017-06-23");
        Date("2017");           // day and month is set to 01 Jan
        Date("June 23 2017");   // long date format
        Date("2017-06-23T12:00:00-09:45");  // date - time YYYY-MM-DDTHH:MM:SSZ
        Date("Jun 23 2017 07:45:00 GMT+0100 (Tokyo Time)"); // time zone

    get time from a date 
        d.getDay();           // getting the weekday
            getDate();          // day as a number (1-31)
            getFullYear();      // four digit year (yyyy)
            getHours();         // hour (0-23)
            getMilliseconds();  // milliseconds (0-999)
            getMinutes();       // minutes (0-59)
            getMonth();         // month (0-11)
            getSeconds();       // seconds (0-59)
            getTime();          // milliseconds since 1970

    get/set time from a date 
        d.setDate(d.getDate() + 7); // adds a week to a date
            setDate();                // day as a number (1-31)
            setFullYear();            // year (optionally month and day)
            setHours();               // hour (0-23)
            setMilliseconds();        // milliseconds (0-999)
            setMinutes();             // minutes (0-59)
            setMonth();               // month (0-11)
            setSeconds();             // seconds (0-59)
            setTime();                // milliseconds since 1970
